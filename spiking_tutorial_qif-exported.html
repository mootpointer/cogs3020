import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

# simulate a single quadratic integrate-and-fire neuron
tau = 1.0
C = 25
vr = -60
vt = -40
k = 0.7
c = -50
vpeak = 35

n_steps = 900

input_amp = 200
E = np.concatenate((np.zeros(n_steps // 3), input_amp * np.ones(n_steps // 3),
                    np.zeros(n_steps // 3)))

v = np.zeros(n_steps)
v[0] = vr

alpha = 0.01
beta = 1.0
o = np.zeros(n_steps)

for i in range(n_steps - 1):
    v[i + 1] = v[i] + tau * (k * (v[i] - vr) * (v[i] - vt) + E[i]) / C
    o[i + 1] = o[i] + alpha * (np.heaviside(v[i] - vt, vt) - o[i])
    if v[i + 1] >= vpeak:
        v[i] = vpeak
        v[i + 1] = c

fig, ax = plt.subplots(nrows=2, ncols=1)
ax[0].plot(np.arange(0, n_steps, 1), E)
ax[0].set_xlabel('Time')
ax[0].set_ylabel('Input Current')

ax[1].plot(np.arange(0, n_steps, 1), v, 'C0')
ax[1].set_xlabel('Time')
ax[1].set_ylabel('Membrane Potential')
ax1 = ax[1].twinx()
ax1.plot(np.arange(0, n_steps, 1), o, 'C1')
ax1.set_ylabel('Output')
ax1.yaxis.label.set_color('C1')

plt.tight_layout()
plt.show()
